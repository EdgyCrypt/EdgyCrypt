In HTML to link assets please use the following Templates

<script type="text/javascript", src="{{ ASSET_URL }}"></script>
{% endassets %}

{% assets "index_css" %}
<link rel="stylesheet" href="{{ ASSET_URL }}"/>
{% endassets %}
